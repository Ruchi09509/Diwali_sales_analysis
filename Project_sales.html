# Group by 'State', sum orders, sort in descending order, and select top 10 states
sales_state = df.groupby('State', as_index=False)['Orders'].sum().sort_values(by='Orders', ascending=False).head(10)

# Set figure size
sns.set(rc={'figure.figsize': (15, 5)})

# Create histogram-like bar plot
ax = sns.histplot(data=sales_state, x='State', weights='Orders', bins=len(sales_state), 
                   discrete=True, stat="count", color='royalblue')

# Annotate each bar with the order value
for p in ax.patches:
    ax.annotate(f"{int(p.get_height())}", 
                (p.get_x() + p.get_width() / 2, p.get_height()), 
                ha='center', va='bottom', fontsize=12, fontweight='bold')

# Add labels and title
plt.xlabel("State", fontsize=14)
plt.ylabel("Total Orders", fontsize=14)
plt.title("Histogram of Total Orders from Top 10 States", fontsize=16, fontweight='bold')

# Rotate x-axis labels for better readability
plt.xticks(rotation=45, fontsize=12)

# Show plot
plt.show()

sales_state = df.groupby(['Product_ID'], as_index=False)['Orders'].sum().sort_values(by='Orders', ascending=False).head(5)

# Define labels and values
labels = sales_state['Product_ID']
sizes = sales_state['Orders']

# Set figure size
plt.figure(figsize=(10, 6))

# Create pie chart
plt.pie(sizes, labels=labels, autopct='%1.1f%%', startangle=140, 
        colors=plt.cm.viridis(np.linspace(0, 1, len(labels))), wedgeprops={'edgecolor': 'black'})
